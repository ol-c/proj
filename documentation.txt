datatypes:

    implemented:
        String
        Number
        File
        Boolean
        Null
        Undefined
        Date
        Function
        Error
        File
        Promise
        Hashmap

    not implemented:
        List
        RegExp

global objects:

    implemented:
        load    //  loads a module into the current context
        unload  //  explicitly unloads a module
        resolve //  resolve a reference
        File    //  get a promise for a remote file
        Promise //  Promise object
        html    //  helps generate live html
        type    //  get the type of an object, must be one of the above datatypes
        local   //  reference to the root object of the current scope
        remote  //  reference to any object within ol-c

    not implemented:
        delay   //  creates a delay before the execution of a function

object special fields:

    implemented
        render       //  returns a string or a promise for a string for the front end to evaluate
        permissions  //  sets the access rules for an object
